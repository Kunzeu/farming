# Patreon OAuth Configuration
# IMPORTANTE: Reemplaza los valores con tus credenciales reales de Patreon

PATREON_CLIENT_SECRET=tu_client_secret_aqui

# Para desarrollo local

NEXT_PUBLIC_PATREON_CLIENT_ID=tu_client_id_aqui
NEXT_PUBLIC_PATREON_REDIRECT_URI=http://localhost:3000/auth/patreon/callback
